% Figure: Normal Exercise Response
% Healthy adaptation to physical activity with rapid recovery

\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=0.75, every node/.style={scale=0.75},
    % Styles
    process/.style={draw=green!70!black, fill=green!10, very thick, rounded corners, text width=4.5cm, align=center, minimum height=1.1cm},
    output/.style={draw=green!70!black, fill=green!25, very thick, rounded corners, text width=4.5cm, align=center, minimum height=1.1cm},

    note/.style={font=\small\itshape, text width=3.5cm, align=center},
]

% Title
\node[font=\large\bfseries, green!70!black] at (0, 9) {Normal Exercise Response};

% Exercise
\node[process] (exercise) at (0, 7.5) {\textbf{Physical Activity}\\(within capacity)};

% Metabolic response
\node[process] (atp) at (0, 5.5) {\textbf{ATP Utilization}\\Glycogen depletion\\Minor lactate production};
\draw[arrow] (exercise) -- (atp);

% Adaptive signaling
\node[process] (signaling) at (0, 3.5) {\textbf{Adaptive Signaling}\\PGC-1$\alpha$, AMPK activation\\Mitochondrial biogenesis};
\draw[arrow] (atp) -- (signaling);
\node[note, right=0.8cm of signaling] {Hormesis:\\beneficial stress\\response};

% Recovery
\node[process] (recovery) at (0, 1.5) {\textbf{Rapid Recovery}\\ATP restored (hours)\\Glycogen restored (1 day)};
\draw[arrow] (signaling) -- (recovery);

% Positive outcome
\node[output] (outcome) at (0, -0.7) {\textbf{NET BENEFIT}\\Increased capacity\\Improved fitness\\Positive adaptation};
\draw[arrow] (recovery) -- (outcome);

% Key point box
\node[draw=green!70!black, fill=green!5, rounded corners, text width=10cm, align=left, font=\small] at (0, -3.2) {
\textbf{Normal exercise adaptation:} Temporary fatigue triggers beneficial adaptations. Recovery occurs within hours to 1 day. Each exercise bout increases capacity---this is the training effect that makes healthy people stronger.
};
