% Figure: Normal Oxidative Stress Balance
% Healthy balance between ROS production and antioxidant defense

\begin{figure}[htbp]

\begin{tikzpicture}[scale=0.75, every node/.style={scale=0.75},
    % Styles
    process/.style={draw=green!70!black, fill=green!10, very thick, rounded corners, text width=4cm, align=center, minimum height=1.2cm},
    output/.style={draw=green!70!black, fill=green!25, very thick, rounded corners, text width=4cm, align=center, minimum height=1.2cm},

% Title
\node[font=\large\bfseries, green!70!black] at (0, 8) {Normal Oxidative Stress Balance};

% ROS production (left)
\node[process] (ros) at (-3.5, 5.5) {\textbf{ROS Production}\\2\% ETC leakage\\(normal byproduct)};

% Antioxidant defense (right)
\node[process] (antioxidants) at (3.5, 5.5) {\textbf{Antioxidant Defense}\\SOD, Catalase, GPx\\Glutathione system};

% Balance point
\node[output] (balance) at (0, 3) {\textbf{ROS BALANCE}\\Production = Neutralization\\No net oxidative damage};

\draw[arrow] (ros) -- (balance);
\draw[arrow] (antioxidants) -- (balance);

% Minor damage and repair
\node[process] (repair) at (0, 0.8) {\textbf{Minor Damage}\\Rapid repair\\Maintained function};
\draw[arrow] (balance) -- (repair);

% Outcome
\node[output] (outcome) at (0, -1.5) {\textbf{Cellular Homeostasis}\\Healthy membranes, proteins, DNA};
\draw[arrow] (repair) -- (outcome);

% Key point box
\node[draw=green!70!black, fill=green!5, rounded corners, text width=10cm, align=left, font=\small] at (0, -4) {
\textbf{Key antioxidants:}
\begin{itemize}[leftmargin=*, nosep, topsep=2pt]
\item \textbf{SOD} (superoxide dismutase) -- converts superoxide to H\textsubscript{2}O\textsubscript{2}
\item \textbf{Catalase} -- converts H\textsubscript{2}O\textsubscript{2} to water
\item \textbf{Glutathione (GSH)} -- master antioxidant, regenerated by GPx
\item \textbf{Vitamin E/C, CoQ10} -- lipid protection, ROS scavenging
\end{itemize}
};
