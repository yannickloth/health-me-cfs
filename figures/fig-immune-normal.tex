% Figure: Normal Immune Response
% Balanced activation, clearance, and resolution

\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=0.75, every node/.style={scale=0.75},
    % Styles
    process/.style={draw=green!70!black, fill=green!10, very thick, rounded corners, text width=4.5cm, align=center, minimum height=1.1cm},
    output/.style={draw=green!70!black, fill=green!25, very thick, rounded corners, text width=4.5cm, align=center, minimum height=1.1cm},
    arrow/.style={-latex, very thick, green!70!black},
    note/.style={font=\small\itshape, text width=3.5cm, align=center},
]

% Title
\node[font=\large\bfseries, green!70!black] at (0, 9) {Normal Immune Response};

% Pathogen exposure
\node[process] (pathogen) at (0, 7.5) {\textbf{Pathogen Exposure}\\Virus, bacteria, etc.};

% Innate response
\node[process] (innate) at (0, 5.5) {\textbf{Innate Response}\\NK cells, macrophages\\Measured cytokine release};
\draw[arrow] (pathogen) -- (innate);

% Adaptive response
\node[process] (adaptive) at (0, 3.5) {\textbf{Adaptive Response}\\T-cell activation\\B-cell antibody production};
\draw[arrow] (innate) -- (adaptive);

% Pathogen clearance
\node[process] (clearance) at (0, 1.5) {\textbf{Pathogen Clearance}\\Effective elimination\\Memory cells formed};
\draw[arrow] (adaptive) -- (clearance);
\node[note, right=0.8cm of clearance] {Immune memory\\protects against\\future infection};

% Resolution
\node[output] (resolution) at (0, -0.7) {\textbf{Resolution}\\Anti-inflammatory signals\\Inflammation stops\\Full recovery};
\draw[arrow] (clearance) -- (resolution);

% Key point box
\node[draw=green!70!black, fill=green!5, rounded corners, text width=10cm, align=left, font=\small] at (0, -3.2) {
\textbf{Normal immune cycle:} Activation $\rightarrow$ Response $\rightarrow$ Clearance $\rightarrow$ Resolution. Anti-inflammatory signals (IL-10, TGF-$\beta$, resolvins) terminate inflammation after pathogen elimination. The system returns to baseline, ready for the next challenge.
};

\end{tikzpicture}
\caption{Normal immune response with appropriate activation and resolution.}
\label{fig:immune-normal}
\end{figure}
