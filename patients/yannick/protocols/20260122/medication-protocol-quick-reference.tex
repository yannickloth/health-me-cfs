\begin{multicols}{3}

%------------------------------------------------------------------------------
\section*{Morning (with/before breakfast)}

\subsection*{Stimulants (max 3 pills/day total)}
\begin{itemize}
    \item \textbf{Rilatine MR 30\,mg} (1--2$\times$/day)
    \item \textbf{Provigil 100\,mg} (1--2$\times$/day)
    \item Can combine (e.g., 1 Rilatine + 1 Provigil)
    \item Avoid late-day dosing (sleep disruption)
\end{itemize}

\subsection*{MCAS/Mast Cell/Allergy Protocol}
\begin{itemize}
    \item \textbf{Rupatadine 10--20\,mg} (H1 + PAF + mast cell stabilizer)
    \item \textbf{Famotidine 20\,mg} (H2 blocker)
    \item \textbf{Quercetin 500--1000\,mg} (natural mast cell stabilizer)
\end{itemize}

\textbf{Indication:} Food allergies (soy, nuts, eggs), pollen allergies, brain fog, GI symptoms, joint pain

\textbf{Dosing:} Start 10\,mg rupatadine; increase to 20\,mg during pollen season (March--Sep) or if insufficient benefit

\subsection*{Core Supplements (take together)}
\begin{itemize}
    \item \textbf{LDN 3\,mg} (immune modulation)
    \item \textbf{Acetyl-L-Carnitine 1000\,mg} (empty stomach preferred)
    \item \textbf{Urolithin A 2000\,mg + NAD+ 200\,mg} (2 caps)
    \item \textbf{CoQ10 Ubiquinol 100\,mg} \textcolor{red}{$\star$ needs fat}
    \item \textbf{NAC 600\,mg} (N-Acétylcystéine, Lysomucil -- glutathion precursor)
    \item \textbf{BEFACT FORTE} (1 tablet, B-complex)
    \item \textbf{Vitamin C 500\,mg}
    \item \textbf{Electrolytes 250\,mL} (7\,g dry mix)
    \item \textbf{FerroDyn FORTE} (1 cap)
\end{itemize}

\subsection*{Breakfast Requirements}
\begin{tcolorbox}[colback=yellow!10,colframe=orange!75!black,boxrule=0.5pt]
\textbf{Must contain dietary fat for CoQ10 absorption:}\\
Eggs, cheese, butter, nuts, olive oil, or MCT oil
\end{tcolorbox}

\subsection*{Iron Optimization (optional)}
For better absorption: Take iron \textbf{1 hour before breakfast} with only Vitamin C 500\,mg. Avoid coffee/tea for 1 hour.

%------------------------------------------------------------------------------
\section*{Afternoon}

\begin{itemize}
    \item \textbf{Electrolytes 250\,mL} (7\,g dry mix)
    \item Optional 2nd stimulant dose if needed
\end{itemize}

\textbf{Rationale:} Clears lactic acid, maintains blood volume, provides glucose when fat-burning impaired

%------------------------------------------------------------------------------
\section*{Midday/Lunch or Evening/Dinner}

\subsection*{Riboflavin B2}
\begin{itemize}
    \item \textbf{Riboflavin 400\,mg} \textcolor{red}{$\star$ needs fat}
    \item Take at \textbf{either} lunch \textbf{or} dinner (choose meal with more fat)
\end{itemize}

\subsection*{Weekly (with dinner)}
\begin{itemize}
    \item \textbf{D-Cure 25000\,U.I.} (Vitamin D3) \textcolor{red}{$\star$ needs fat}
    \item Once weekly with fatty meal
\end{itemize}

\subsection*{Evening (6--8 hours after breakfast)}
\begin{itemize}
    \item \textbf{Famotidine 20\,mg} (second H2 blocker dose for MCAS)
\end{itemize}

\textbf{Note:} Famotidine taken morning + evening provides 24-hour mast cell control

\subsection*{Meal Requirements}
\begin{tcolorbox}[colback=yellow!10,colframe=orange!75!black,boxrule=0.5pt]
\textbf{Must contain dietary fat:}\\
Fatty fish, olive oil (salad dressing), avocado, cheese, nuts, or MCT oil
\end{tcolorbox}

%------------------------------------------------------------------------------
\section*{Bedtime}

\subsection*{Core Bedtime Protocol}
\begin{itemize}
    \item \textbf{MCT Oil 1 tsp (5\,mL)} (30--60 min before bed)
    \item \textbf{D-Ribose 5\,g} (dissolved in water)
    \item \textbf{Magnesium Glycinate 300--400\,mg}
    \item Optional: small slow-carb snack if severe cramps
\end{itemize}

\subsection*{Timing Rule}
\begin{tcolorbox}[colback=red!10,colframe=red!75!black,boxrule=0.5pt]
\textbf{Minimum 2--4 hours after last stimulant}\\
(ideally 6--8 hours separation for magnesium)
\end{tcolorbox}

%------------------------------------------------------------------------------
\section*{MCAS/Mast Cell Activation Management}

\subsection*{Complete Protocol (Year-Round + Seasonal)}
\begin{tcolorbox}[colback=blue!10,colframe=blue!75!black,boxrule=0.5pt]
\textbf{Year-Round Baseline (October--February):}
\begin{itemize}
    \item Rupatadine 10\,mg morning
    \item Famotidine 20\,mg twice daily (morning + evening)
    \item Quercetin 500--1000\,mg morning
    \item Low-histamine diet awareness (avoid aged/fermented foods)
\end{itemize}

\textbf{Pollen Season Intensification (March--September):}
\begin{itemize}
    \item Rupatadine \textbf{20\,mg} morning (double dose)
    \item Famotidine 20\,mg twice daily (unchanged)
    \item Quercetin 1000\,mg morning (max dose)
    \item Stricter low-histamine diet
\end{itemize}
\end{tcolorbox}

\subsection*{Why Rupatadine is Superior}
\begin{itemize}
    \item Triple mechanism: H1 antagonist + PAF antagonist + direct mast cell stabilizer
    \item 31$\times$ more potent than loratadine at PAF antagonism
    \item Highest efficacy rank (SUCRA 99.7\%) vs all other antihistamines
    \item Addresses vascular dysfunction and orthostatic intolerance via PAF blockade
\end{itemize}

\subsection*{Expected Benefits (2--4 weeks)}
\begin{itemize}
    \item Brain fog reduction, improved cognitive clarity
    \item Energy improvement (especially post-meal fatigue)
    \item GI symptom relief (nausea, bloating)
    \item Better orthostatic tolerance
    \item Reduced flushing, allergic symptoms, anxiety episodes
\end{itemize}

\subsection*{Safe Combinations}
No interactions with: methylphenidate, modafinil, LDN, mitochondrial supplements

\textbf{Iron timing:} Take iron 2 hours before/after famotidine (H2 blocker reduces acid)

    \item \textbf{MCT Oil}: Start 1 tsp daily, increase to 1 tbsp over 1--2 weeks
    \item Take with fat-soluble vitamins (morning or evening)
    \item Add to coffee, tea, smoothies, or food
    \item \textcolor{red}{Increase slowly to avoid diarrhea}

\subsection*{Digestive Enzymes}
\begin{itemize}
    \item \textbf{Metagenics MetaDigest TOTAL} (comprehensive enzyme formula)

%------------------------------------------------------------------------------
\section*{Critical Separations}

\begin{tcolorbox}[colback=red!10,colframe=red!75!black,boxrule=1pt,title=\textbf{DO NOT take within 2--4 hours of Rilatine MR}]
\begin{itemize}
    \item Magnesium carbonate/oxide/hydroxide
    \item Calcium carbonate (Tums)
    \item Sodium bicarbonate (baking soda)
    \item Antacids (Gaviscon, Rennie, etc.)
\end{itemize}
\textbf{Why:} Premature methylphenidate release $\rightarrow$ heart rate spikes, reduced duration
\end{tcolorbox}

\begin{tcolorbox}[colback=green!10,colframe=green!75!black,boxrule=0.5pt,title=\textbf{Safe near stimulants:}]
Electrolyte solution, magnesium glycinate (at bedtime only), food
\end{tcolorbox}

\subsection*{Iron $\leftrightarrow$ Calcium/Magnesium}
Separate by 2--4 hours for optimal iron absorption (compete for same transporters)

\section*{Electrolyte Solution Protocol}

\subsection*{Dry Mix Preparation (Standard Formula)}
\begin{itemize}
    \item 100\,g white sugar
    \item 15\,g Jozo low-sodium salt (potassium source)
    \item 15\,g table salt (sodium source)
    \item \textbf{Total: 130\,g}
\end{itemize}

\subsection*{Per-Dose Preparation (2$\times$/day)}
\begin{itemize}
    \item 7\,g dry mix in 250\,mL water
    \item 10\,mL grenadine (or sugar-free alternative)
    \item Sip throughout day
\end{itemize}

\subsection*{Lower-Sugar Alternative}
\begin{itemize}
    \item Use \textbf{50\,g sugar} (instead of 100\,g)
    \item Same salt amounts (15\,g each)
    \item Total: 80\,g dry mix
    \item Use 4.3\,g per 250\,mL dose
    \item Use sugar-free grenadine or lemon
    \item 80\% sugar reduction, same electrolyte benefit
\end{itemize}

\section*{Important Reminders}

\subsection*{Stimulant Warning}
\begin{tcolorbox}[colback=red!10,colframe=red!75!black,boxrule=1pt]
\textbf{Stimulants mask true energy levels}\\
Trust heart rate monitor, not feelings\\
Maximum 3 pills/day total across both medications
\end{tcolorbox}

\subsection*{Heart Rate Pacing}
\textbf{Target HR limit:} $(220 - \text{age}) \times 0.55$\\
Stay below this to avoid lactic acid buildup

\subsection*{False Energy Risk}
Both methylphenidate and modafinil allow ``borrowing'' from depleted reserves. Heart rate monitoring essential.

\subsection*{Migraine Interaction}
Both stimulants cause vasoconstriction (migraine trigger). Makes riboflavin 400\,mg/day and hydration critical.
