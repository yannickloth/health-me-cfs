\subsection*{Stimulants (Max 3 pills/day total)}

\textbf{Rilatine MR 30\,mg:} Dopamine/norepinephrine boost for cognitive function. \textcolor{red}{NEVER with magnesium carbonate/antacids} (causes dose dumping). Masks energy---use HR monitor.

\textbf{Provigil 100\,mg:} Wakefulness support. \textcolor{green}{No supplement interactions}. Also masks energy.

\subsection*{MCAS/Mast Cell/Allergy Management}

\textbf{Rupatadine 10--20\,mg (morning):} \textcolor{blue}{TRIPLE MECHANISM}: H1 antagonist + PAF antagonist + direct mast cell stabilizer. 31$\times$ more potent than loratadine. Highest efficacy rank (SUCRA 99.7\%). Targets brain fog, energy, allergies, orthostatic intolerance. Year-round 10\,mg; increase to 20\,mg pollen season (Mar--Sep).

\textbf{Famotidine 20\,mg BID (morning + evening):} H2 blocker. Essential for combination therapy (H1 alone ineffective). Provides 24-hour mast cell control. Safe long-term. Note: May reduce stomach acid---take iron 2 hours apart.

\textbf{Quercetin 500--1000\,mg (morning):} Natural mast cell stabilizer. MORE effective than prescription cromolyn in studies. Anti-inflammatory, antioxidant. OTC, well-tolerated. 500\,mg maintenance, 1000\,mg pollen season.

\textbf{Indication:} Food allergies (soy IgG 88 mg/L, nuts, eggs), pollen allergies (grass 8.89 kUA/L), brain fog, GI symptoms, joint pain, flushing.

\textbf{Timeline:} Week 1: reduced allergic symptoms. Weeks 2--4: brain fog, energy, GI relief (if MCAS-related). Week 4: discontinuation test confirms benefit.

\subsection*{Immune \& Core Energy}

\textbf{LDN 3\,mg:} Immune modulation, anti-inflammatory. 4--12 weeks for effect.

\textbf{Acetyl-L-Carnitine 1000\,mg:} \textcolor{blue}{ROOT CAUSE FIX}. Opens carnitine shuttle for fat-burning. 4--6 weeks initial, 3--6 months maximum.

\textbf{CoQ10 Ubiquinol 100\,mg:} Electron transport chain ``spark plug.'' \textcolor{red}{Needs dietary fat} or <10\% absorption.

\textbf{Riboflavin B2 400\,mg:} Fat-burning cofactor. Migraine prevention. \textcolor{red}{Needs dietary fat}. 4--12 weeks for migraines.

\textbf{Urolithin A + NAD+:} Mitochondrial health (mitophagy + cofactor priming).

\subsection*{Vitamins \& Minerals}

\textbf{D-Cure 25000\,U.I. (weekly):} Immune, bone, mood. \textcolor{red}{MUST have dietary fat} or stays deficient. Chronic deficiency indicates fat malabsorption.

\textbf{BEFACT FORTE:} B-complex for energy metabolism cofactors.

\textbf{Vitamin C 500\,mg:} Antioxidant, enhances iron absorption.

\textbf{Magnesium Glycinate 300--400\,mg (bedtime):} Muscle relaxation (``off switch''), ATP synthesis cofactor, nocturnal cramp prevention. Glycinate form safe 6--8 hrs after stimulants.

\textbf{FerroDyn FORTE:} Iron for oxygen transport. Best with vitamin C, separate from calcium/magnesium by 2--4 hours. Avoid coffee/tea for 1 hour.

\textbf{Vitamin A 5000\,IU:} Vision support (rhodopsin regeneration). \textcolor{red}{Needs dietary fat}.

\subsection*{Vision Support}

\textbf{Lutein 10--20\,mg + Zeaxanthin 2--4\,mg:} Macular protection, blue light filter. \textcolor{red}{Need fat}.

\textbf{Taurine 500--1000\,mg:} Retinal cell function (water-soluble).

\textbf{DHA 500--1000\,mg:} Retinal membrane structure. \textcolor{red}{Needs fat}.

\subsection*{Electrolytes}

\textbf{Custom solution 250\,mL, 2$\times$/day:} Sodium expands blood volume, potassium enables muscle relaxation, glucose provides quick energy when fat-burning impaired. Afternoon dose clears lactic acid.

\subsection*{Fat Malabsorption Support}

\textbf{MCT Oil 1 tbsp/day:} \textcolor{blue}{IMMEDIATE ENERGY BYPASS}. C8/C10 fats bypass broken carnitine shuttle. Go straight to liver for energy. Support fat-soluble vitamin absorption. Start 1 tsp, increase slowly. Take 1 tsp before bed for nocturnal ATP support.

\textbf{MetaDigest TOTAL (Metagenics):} Comprehensive enzyme formula (lipase, protease, amylase, cellulase, lactase). Compensates for low pancreatic enzyme production. Take immediately before meals with fat-soluble vitamins.

\subsection*{Nocturnal ATP Depletion}

\textbf{D-Ribose 5\,g (bedtime + morning):} Direct ATP building block. Replenishes ATP overnight when fat-burning blocked. Reduces nocturnal cramps, morning exhaustion. 2 weeks to assess.

\begin{tcolorbox}[colback=red!10,colframe=red!75!black,boxrule=0.5pt,title=\textbf{NEVER Together}]
\textbf{Rilatine MR + Magnesium carbonate/oxide/antacids}\\
Causes premature drug release $\rightarrow$ HR spikes\\
\textbf{Separation:} 2--4 hours minimum (bedtime ideal)
\end{tcolorbox}

\begin{tcolorbox}[colback=green!10,colframe=green!75!black,boxrule=0.5pt]
\textbf{Provigil:} No supplement interactions\\
\textbf{Magnesium glycinate at bedtime:} Safe (6--8 hrs after stimulants)
\end{tcolorbox}

\textbf{Iron $\leftrightarrow$ Calcium/Magnesium:} Separate 2--4 hours (compete for absorption).

\subsection*{Fat-Soluble (MUST have fat)}

CoQ10, Riboflavin B2, Vitamin D3, Vitamin A, Lutein, Zeaxanthin, DHA

\textcolor{red}{\textbf{Without fat: <10\% absorption!}}

\textbf{Good fats:} MCT oil, olive oil, fatty fish, avocado, nuts, eggs

\subsection*{The Energy Strategy}

\begin{tcolorbox}[colback=blue!5,colframe=blue!75!black,boxrule=0.5pt]
\textbf{Root Problem:} Blocked carnitine shuttle prevents fat-burning

\textbf{3-Part Solution:}

\textbf{1. Immediate Bypass} (MCT oil + D-Ribose)
\begin{itemize}[nosep,leftmargin=0.3cm]
    \item MCT oil bypasses broken shuttle
    \item D-Ribose replenishes ATP directly
    \item Effect: Immediate energy, reduced cramps
\end{itemize}

\textbf{2. Root Repair} (Acetyl-L-Carnitine)
\begin{itemize}[nosep,leftmargin=0.3cm]
    \item Opens carnitine shuttle (4--6 weeks)
    \item Restores fat metabolism (3--6 months max)
\end{itemize}

\textbf{3. System Optimization} (CoQ10, B2, Mg)
\begin{itemize}[nosep,leftmargin=0.3cm]
    \item CoQ10: electron transport chain
    \item Riboflavin: fat-burning cofactor
    \item Magnesium: ATP synthesis + muscle relaxation
\end{itemize}

\textbf{Fat Malabsorption Fix:}
\begin{itemize}[nosep,leftmargin=0.3cm]
    \item MCT oil + enzymes ensure vitamin absorption
    \item Fixes chronic vitamin D deficiency
\end{itemize}
\end{tcolorbox}

\subsection*{Timelines}

\textbf{Days 1--7:} MCT + D-Ribose immediate effect

\textbf{Weeks 4--6:} Acetyl-L-carnitine begins opening shuttle

\textbf{Months 2--3:} Retest vitamin D levels

\textbf{Months 3--6:} Full fat-burning restoration

\textbf{Riboflavin migraines:} 4--12 weeks

\subsection*{Key Warnings}

\begin{itemize}[nosep,leftmargin=0.3cm]
    \item \textbf{Stimulants mask energy:} Trust HR monitor, not feelings
    \item \textbf{Max 3 stimulant pills/day:} Both medications combined
    \item \textbf{Fat essential:} D3, CoQ10 won't absorb without it
    \item \textbf{MCT oil:} Start slow (1 tsp) to avoid diarrhea
    \item \textbf{Coconut oil $\neq$ MCT oil:} Only pure C8/C10 works
    \item \textbf{HR pacing:} $(220 - \text{age}) \times 0.55$ to avoid lactic acid
\end{itemize}

\section*{Stimulant Medications --- For Wakefulness \& Cognitive Function}
%------------------------------------------------------------------------------

\subsection*{Rilatine MR (Methylphenidate) 30\,mg}
\textbf{What it does:}
\begin{itemize}
    \item Increases dopamine and norepinephrine in the brain
    \item Improves alertness, focus, and cognitive function
    \item Helps counteract severe fatigue and brain fog
\end{itemize}

\textbf{Why you're taking it:}
ME/CFS causes severe cognitive impairment and fatigue. This medication provides temporary cognitive support to maintain function.

\textbf{Important:}
\begin{itemize}
    \item Modified-release formulation --- \textcolor{red}{NEVER with magnesium carbonate/antacids} (causes dangerous ``dose dumping'')
    \item Masks true energy levels --- trust heart rate monitor, not feelings
    \item Can trigger migraines (vasoconstriction) --- riboflavin helps prevent this
    \item Maximum 3 pills/day total (combined with Provigil)
\end{itemize}

\subsection*{Provigil (Modafinil) 100\,mg}
\textbf{What it does:}
\begin{itemize}
    \item Promotes wakefulness through different mechanism than methylphenidate
    \item Improves sustained alertness and reduces ``sleepiness''
    \item Less stimulating than methylphenidate but longer-lasting
\end{itemize}

\textbf{Why you're taking it:}
Provides complementary wakefulness support; can be used alone or combined with methylphenidate depending on daily needs.

\textbf{Important:}
\begin{itemize}
    \item \textcolor{green}{No significant supplement interactions} --- safe to take with everything
    \item Also masks true energy levels --- heart rate monitoring essential
    \item Can trigger migraines (vasoconstriction) --- riboflavin helps
    \item Maximum 3 pills/day total (combined with Rilatine)
\end{itemize}

%------------------------------------------------------------------------------
\section*{Immune Modulation}
%------------------------------------------------------------------------------

\subsection*{LDN (Low-Dose Naltrexone) 3\,mg}
\textbf{What it does:}
\begin{itemize}
    \item Modulates immune system by temporarily blocking opioid receptors
    \item Reduces inflammation throughout the body
    \item May improve energy and reduce pain over time
\end{itemize}

\textbf{Why you're taking it:}
ME/CFS involves immune dysfunction and chronic inflammation. LDN helps rebalance the immune system.

\textbf{Timeline:} 4--12 weeks to assess effectiveness. Started 2026-01-05, increasing to 4--4.5\,mg after current stock.

%------------------------------------------------------------------------------
\section*{Mitochondrial Support --- The Core Energy System}
%------------------------------------------------------------------------------

\subsection*{Acetyl-L-Carnitine 1000\,mg}
\textbf{What it does:}
\begin{itemize}
    \item Opens the ``carnitine shuttle'' --- transports long-chain fats into mitochondria for energy production
    \item Acetyl form crosses blood-brain barrier for cognitive support
    \item Addresses root cause of fat metabolism dysfunction
\end{itemize}

\textbf{Why you're taking it:}
\begin{itemize}
    \item You cannot burn stored body fat for energy (carnitine shuttle blocked)
    \item This causes ``running on empty'' sensation despite adequate calorie intake
    \item Also targets muscle cramps (ATP depletion from failed fat oxidation)
    \item Supports cognitive function (brain energy metabolism)
\end{itemize}

\textbf{Timeline:} 4--6 weeks for initial effect; 3--6 months for maximum benefit. This is the \textbf{root cause fix}.

\subsection*{CoQ10 Ubiquinol 100\,mg}
\textbf{What it does:}
\begin{itemize}
    \item ``Spark plug'' in the electron transport chain (mitochondrial energy production)
    \item Antioxidant protecting mitochondrial membranes
    \item Essential cofactor for ATP synthesis
\end{itemize}

\textbf{Why you're taking it:}
Mitochondrial dysfunction means inefficient ATP production. CoQ10 supports the energy production machinery.

\textbf{Important:} \textcolor{red}{Fat-soluble} --- MUST take with dietary fat (eggs, olive oil, MCT oil, nuts) or absorption is <10\%.

\subsection*{Riboflavin (Vitamin B2) 400\,mg}
\textbf{What it does:}
\begin{itemize}
    \item Precursor to FAD (flavin adenine dinucleotide) --- essential for beta-oxidation (fat burning)
    \item Critical cofactor in electron transport chain
    \item Migraine prevention (proven at 400\,mg/day)
\end{itemize}

\textbf{Why you're taking it:}
\begin{itemize}
    \item Supports fat metabolism (works with acetyl-L-carnitine)
    \item Prevents migraines triggered by stimulant vasoconstriction
    \item Essential for mitochondrial energy production
\end{itemize}

\textbf{Important:} \textcolor{red}{Fat-soluble} --- take with lunch or dinner containing fat. Migraine prevention takes 4--12 weeks.

\subsection*{Urolithin A + NAD+ (Joiavvy) 2 capsules}
\textbf{What it does:}
\begin{itemize}
    \item Urolithin A: Promotes mitophagy (removal of damaged mitochondria) and mitochondrial biogenesis (creating new mitochondria)
    \item NAD+: Cofactor that ``primes'' the energy cycle; declines with age and disease
\end{itemize}

\textbf{Why you're taking it:}
Supports mitochondrial health and function at the cellular level.

\section*{Vitamins \& Minerals}
%------------------------------------------------------------------------------

\subsection*{D-Cure 25000\,U.I. (Vitamin D3) --- Weekly}
\textbf{What it does:}
\begin{itemize}
    \item Supports immune function, bone health, mood regulation
    \item Anti-inflammatory effects
    \item Essential for calcium absorption and cellular function
\end{itemize}

\textbf{Why you're taking it:}
\textbf{Chronic vitamin D deficiency for years} despite daily 3000\,U.I.\ supplementation. Strongly suggests fat malabsorption (common in ME/CFS).

\textbf{Important:}
\begin{itemize}
    \item \textcolor{red}{Fat-soluble} --- MUST take with fatty meal (fish, olive oil, avocado, cheese) or will remain deficient
    \item Weekly high-dose may overcome absorption issues better than daily low-dose
    \item Needs follow-up labs in 2--3 months to verify effectiveness
\end{itemize}

\subsection*{BEFACT FORTE (B-Complex)}
\textbf{What it does:}
\begin{itemize}
    \item Provides full spectrum of B vitamins
    \item B vitamins are cofactors in energy metabolism, neurotransmitter synthesis, and methylation
\end{itemize}

\textbf{Why you're taking it:}
Supports energy production and neurological function. Complements riboflavin B2.

\subsection*{Vitamin C 500\,mg}
\textbf{What it does:}
\begin{itemize}
    \item Antioxidant support
    \item Enhances iron absorption (critical synergy)
    \item Supports immune function and collagen synthesis
\end{itemize}

\textbf{Why you're taking it:}
Taken with iron to maximize iron absorption for addressing iron deficiency.

\subsection*{Magnesium Glycinate 300--400\,mg (Bedtime)}
\textbf{What it does:}
\begin{itemize}
    \item ``Off switch'' for muscle contraction --- enables muscle relaxation
    \item Critical cofactor for 300+ enzymatic reactions including ATP synthesis
    \item Supports sleep quality
\end{itemize}

\textbf{Why you're taking it:}
\begin{itemize}
    \item Prevents nocturnal muscle cramps (low ATP causes muscle ``lock-up'')
    \item Supports energy production (cofactor for PDH and TCA cycle)
    \item Taken at bedtime to target overnight cramps when ATP is lowest
\end{itemize}

\textbf{Important:}
\begin{itemize}
    \item Glycinate form has minimal pH effect (safe for bedtime, 6--8 hours after stimulants)
    \item \textcolor{red}{Never use magnesium carbonate/oxide} --- causes methylphenidate dose dumping
\end{itemize}

\subsection*{FerroDyn FORTE (Iron)}
\textbf{What it does:}
\begin{itemize}
    \item Essential for hemoglobin production (oxygen transport)
    \item Cofactor in energy metabolism and neurotransmitter synthesis
\end{itemize}

\textbf{Why you're taking it:}
Addresses iron deficiency which worsens fatigue and reduces oxygen delivery to tissues.

\textbf{Important:}
\begin{itemize}
    \item Best absorbed on empty stomach with vitamin C
    \item Separate from calcium/magnesium by 2--4 hours (compete for absorption)
    \item Avoid coffee/tea for 1 hour (tannins block absorption)
\end{itemize}

\subsection*{Vitamin A 5000\,IU (to start)}
\textbf{What it does:}
\begin{itemize}
    \item Essential for rhodopsin regeneration (night vision)
    \item Supports retinal function and overall eye health
    \item Immune function support
\end{itemize}

\textbf{Why you're taking it:}
Part of vision support protocol targeting energy-dependent vision deterioration.

\textbf{Important:} \textcolor{red}{Fat-soluble} --- take with olive oil or fatty meal.

Supports retinal metabolism and may stabilize vision quality.

\subsection*{DHA (Omega-3) 500--1000\,mg}

\textbf{Why you're taking it:}
Essential for retinal structure; may slow vision deterioration.

\textbf{Important:} \textcolor{red}{Fat-soluble} --- take with fatty meal (or use high-quality fish oil which contains its own fat).

\section*{Electrolyte Management}
%------------------------------------------------------------------------------

\subsection*{Custom Electrolyte Solution (250\,mL, 2$\times$/day)}
\textbf{What it does:}
\begin{itemize}
    \item \textbf{Sodium}: Expands blood volume (``sponge'' effect pulling water into circulation)
    \item \textbf{Potassium}: Enables muscle relaxation; maintains cellular electrical charge
    \item \textbf{Glucose}: Enhances sodium absorption via SGLT1 transporter; provides quick energy when fat-burning is impaired
\end{itemize}

\textbf{Why you're taking it:}
\begin{itemize}
    \item ME/CFS typically involves low blood volume and orthostatic intolerance
    \item Sodium supports blood pressure and circulation
    \item Potassium prevents muscle cramps (the ``off switch'')
    \item Afternoon dose clears accumulated lactic acid from morning activities
    \item Provides glucose when fat metabolism is blocked
\end{itemize}

\textbf{Formula:} 7\,g dry mix (sugar + Jozo low-sodium salt + table salt) in 250\,mL water. Lower-sugar alternative available (4.3\,g per dose).

\section*{Fat Malabsorption Support}
%------------------------------------------------------------------------------

\subsection*{MCT Oil 1 teaspoon--1 tablespoon daily}
\textbf{What it does:}
\begin{itemize}
    \item Medium-chain triglycerides (C8--C10) bypass normal fat digestion
    \item Absorbed directly without requiring bile acids or pancreatic lipase
    \item Go straight to liver for immediate energy production
    \item \textbf{Do NOT require the carnitine shuttle} --- bypass the broken system
\end{itemize}

\textbf{Why you're taking it:}
\begin{itemize}
    \item \textbf{Emergency energy bypass}: Provides fuel mitochondria can actually use right now
    \item Supports absorption of fat-soluble vitamins (D3, CoQ10, B2)
    \item Preferentially oxidized for energy, rarely stored as body fat
    \item Addresses nocturnal ATP depletion (take 1 tsp before bed)
\end{itemize}

\textbf{Important:}
\begin{itemize}
    \item Start with 1 tsp daily, increase to 1 tbsp over 1--2 weeks
    \item \textcolor{red}{Increase slowly} --- rapid escalation causes diarrhea
    \item Take with fat-soluble vitamins or 30--60 min before bed
    \item This is \textbf{NOT coconut oil} --- MCT oil is extracted/concentrated C8/C10 only
\end{itemize}

\subsection*{MetaDigest TOTAL (Metagenics)}
\textbf{What it does:}
\begin{itemize}
    \item Comprehensive enzyme formula: lipase (breaks down fats), protease (proteins), amylase (carbohydrates), cellulase (fiber), lactase (dairy)
    \item Compensates for inadequate pancreatic enzyme production (energy-dependent)
    \item Critical for fat-soluble vitamin absorption (D3, CoQ10, B2)
\end{itemize}

\textbf{Why you're taking it:}
Pancreas requires energy to produce enzymes. Mitochondrial dysfunction reduces enzyme production, impairing digestion and absorption. This is particularly important for fat malabsorption (evidenced by chronic vitamin D deficiency).

\textbf{Important:} Take immediately before or with first bite of meals containing fat-soluble vitamins.

\section*{Nocturnal ATP Depletion Support}
%------------------------------------------------------------------------------

\subsection*{D-Ribose 5\,g (before bed + morning for 10\,g/day total)}
\textbf{What it does:}
\begin{itemize}
    \item Simple sugar that's a direct building block of the ATP molecule
    \item Replenishes cellular ATP stores rapidly
    \item Bypasses complex metabolic pathways
\end{itemize}

\textbf{Why you're taking it:}
\begin{itemize}
    \item During overnight fast, body should burn fat for ATP production
    \item Blocked carnitine shuttle prevents fat oxidation
    \item ATP progressively depletes through the night
    \item Low ATP causes nocturnal cramps and unrefreshing sleep
    \item D-Ribose provides raw material to maintain ATP overnight
\end{itemize}

\textbf{Timeline:} Some notice effect within days; assess at 2 weeks for nocturnal cramp reduction.

\section*{Summary: How It All Works Together}
%------------------------------------------------------------------------------

\begin{tcolorbox}[colback=blue!5,colframe=blue!75!black,title=The Energy Production Strategy]

\textbf{Root Problem:} Blocked carnitine shuttle prevents burning stored body fat for energy.

\textbf{Three-Part Solution:}

\begin{enumerate}
    \item \textbf{Immediate bypass (MCT Oil + D-Ribose):}
    \begin{itemize}
        \item MCT oil provides fat that doesn't need carnitine shuttle
        \item D-Ribose directly replenishes ATP
        \item Effect: Immediate energy support, reduced nocturnal cramps
    \end{itemize}

    \item \textbf{Root cause repair (Acetyl-L-Carnitine):}
    \begin{itemize}
        \item Opens carnitine shuttle over 4--6 weeks
        \item Enables normal fat burning (stored body fat + dietary fat)
        \item Effect: Gradual restoration of fat metabolism
    \end{itemize}

    \item \textbf{System optimization (CoQ10, Riboflavin, Magnesium):}
    \begin{itemize}
        \item Support the mitochondrial machinery once fuel can get in
        \item CoQ10: electron transport chain function
        \item Riboflavin: fat-burning cofactor
        \item Magnesium: ATP synthesis cofactor; muscle relaxation
    \end{itemize}
\end{enumerate}

\textbf{Fat Malabsorption Fix:}
\begin{itemize}
    \item MCT oil + digestive enzymes ensure fat-soluble vitamins actually absorb
    \item Fixes chronic vitamin D deficiency and improves CoQ10/B2 effectiveness
\end{itemize}

\textbf{Vision Support:}
\begin{itemize}
    \item Addresses energy-dependent ciliary muscle fatigue
    \item Structural support (lutein, zeaxanthin, DHA) + metabolic support (taurine, vitamin A)
\end{itemize}

\textbf{Stimulants:}
\begin{itemize}
    \item Temporary cognitive support while metabolic repair happens
    \item Mask true energy --- use heart rate monitoring for safety
\end{itemize}
